^ Adapt ansible role/task to generate currentTimestamp:

```yml
- name: Generate dat file
  vars:
    currentTimestamp: "{{ lookup('pipe', 'date +%Y-%m-%d--%Hh%Mm%Ss') }}"
  template:
    dest: './README-generated-{{ currentDateTime }}.md'
    src: templates/README.md.j2
```
